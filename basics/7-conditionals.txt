package main

import "fmt"

func main() {
	// in GO, conditionals are pretty much the same as in Python
	age := 7

	if age > 7 {
		fmt.Println("You are older than 7\n")
	} else if age == 7 {
		fmt.Printf("You are 7 years old\n")
	} else {
		fmt.Printf("You are younger than 7\n")
	}

	// but unlike Python, in GO we can declare variables right before conditionals
	if newAge := 5; newAge >= 7 {
		fmt.Println("You are older than 7\n")
	} else if newAge == 7 {
		fmt.Printf("You are 7 years old\n")
	} else {
		fmt.Printf("You are younger than 7\n")
	}
}
